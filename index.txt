function initPageCharts(id){
if(id===‘levels’){
const ctx=document.getElementById(‘ch_levels’);if(!ctx)return;
const LEVELS=[‘aspire’,‘good’,‘devA’,‘devB’,‘normal’];
const LABELS=[‘أسباير’,‘جيد’,‘قابل تطور أ’,‘قابل تطور ب’,‘عادي’];
const COLORS=[’#5BB8FF’,’#22C55E’,’#A855F7’,’#EF4444’,’#7A90C0’];
const data=LEVELS.map(l=>DB.players.filter(p=>p.level===l).length);
DB.charts.levels=new Chart(ctx,{type:‘doughnut’,data:{labels:LABELS,datasets:[{data,backgroundColor:COLORS.map(c=>c+‘BB’),borderColor:COLORS,borderWidth:1.5,hoverOffset:6}]},options:{responsive:true,maintainAspectRatio:false,plugins:{legend:{position:‘bottom’,labels:{boxWidth:10,font:{size:10}}}},cutout:‘60%’}});
}
if(id===‘attendance’){
const ctx=document.getElementById(‘ch_att’);if(!ctx)return;
const pcts=DB.players.map(p=>{const tot=DB.sessions.length,pres=DB.sessions.filter(s=>DB.attendance[`${p.id}_${s.id}`]===‘P’).length;return tot?Math.round(pres/tot*100):0;});
DB.charts.att=new Chart(ctx,{type:‘bar’,data:{labels:DB.players.map(p=>p.name.split(’ ‘)[0]),datasets:[{label:‘نسبة الحضور %’,data:pcts,backgroundColor:pcts.map(v=>v>=80?‘rgba(34,197,94,.7)’:‘rgba(239,68,68,.7)’),borderRadius:4}]},options:{responsive:true,maintainAspectRatio:false,plugins:{legend:{display:false}},scales:{y:{grid:{color:‘rgba(255,255,255,.04)’},max:100,ticks:{font:{size:10}}},x:{grid:{display:false},ticks:{font:{size:10}}}}}});
}
if(id===‘competitions’){
const ctx=document.getElementById(‘ch_results’);if(!ctx)return;
const wins=DB.matches.filter(m=>m.home.includes(‘الخور’)?m.hs>m.as:m.as>m.hs).length;
const draws=DB.matches.filter(m=>m.hs===m.as).length;
const losses=DB.matches.length-wins-draws;
DB.charts.res=new Chart(ctx,{type:‘doughnut’,data:{labels:[`انتصارات (${wins})`,`تعادلات (${draws})`,`خسائر (${losses})`],datasets:[{data:[wins,draws,losses],backgroundColor:[‘rgba(34,197,94,.7)’,‘rgba(212,168,32,.7)’,‘rgba(239,68,68,.7)’],borderColor:[’#22C55E’,’#D4A820’,’#EF4444’],borderWidth:1.5,hoverOffset:6}]},options:{responsive:true,maintainAspectRatio:false,plugins:{legend:{position:‘bottom’,labels:{boxWidth:10,font:{size:10}}}},cutout:‘55%’}});
}
}

function toast(msg){
const ex=document.querySelector(’.toast’);if(ex)ex.remove();
const t=document.createElement(‘div’);t.className=‘toast’;
t.innerHTML=`<span style="font-size:17px">${msg.match(/^[\p{Emoji}]/u)?.[0]||'ℹ️'}</span><span>${msg.replace(/^[\p{Emoji}\s]+/u,'').trim()}</span>`;
document.body.appendChild(t);
setTimeout(()=>{t.style.transition=‘opacity .4s’;t.style.opacity=‘0’;},2800);
setTimeout(()=>t.remove(),3300);
}
